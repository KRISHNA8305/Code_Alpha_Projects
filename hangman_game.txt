import random

def play_hangman():
    word_list = ["america", "new york", "california", "india", "japan","africa"]
    chosen_word = random.choice(word_list).upper()
    word_completion = ["_"] * len(chosen_word)
    guessed_letters = []
    tries = 6

    print("Let's play Hangman!")

    while tries > 0 and "_" in word_completion:
        print("\n" + " ".join(word_completion))
        print(f"Guessed letters: {', '.join(guessed_letters)}")
        print(f"Tries left: {tries}")

        guess = input("Guess a letter: ").upper()

        if len(guess) != 1 or not guess.isalpha():
            print("Invalid input. Please enter a single letter.")
            continue
        if guess in guessed_letters:
            print("You already guessed that letter.")
            continue

        guessed_letters.append(guess)

        if guess in chosen_word:
            for i in range(len(chosen_word)):
                if chosen_word[i] == guess:
                    word_completion[i] = guess
            print("Correct guess!")
        else:
            tries -= 1
            print("Incorrect guess!")

    if "_" not in word_completion:
        print("\n" + " ".join(word_completion))
        print("Congratulations! You guessed the word!")
    else:
        print("\nGame Over! The word was: " + chosen_word)


play_hangman()
